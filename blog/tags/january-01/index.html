 function debounce(fn,delay){
    let timer = null
    //返回事件绑定函数
    return function(){
        // fn this
        let context = this
        // fn 参数
        let args = arguments
        // 清除定时器
        clearTimeout(timer)
        timer = setTimeout(()=>{
            fn.apply(context,args)
        },delay)
    }
 }
