function copyAddObj(oldObj){
    let newObj = {}
    for(let key in oldObj){
        newObj[key] = oldObj[key]
    }
    newObj.add = false
    newObj.update = false
    newObj.delete = false
    newObj.check = false
    return newObj
}
function deepCloneObj(oldObj){
    let newObj = {}
    for (let key in oldObj){
        if(typeof oldObj[key] === 'Array' || typeof oldObj[key] === 'Object){
            newObj[key] = deepCloneObj(oldObj[key])
        }else{
            newObj[key] = oldObj[key]
        }
    }
    return newObj
}
